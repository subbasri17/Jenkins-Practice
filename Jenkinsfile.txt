pipeline {
    agent any
    stages {
        stage("Clean Up"){
            steps {
                deleteDir()
            }
        }
        stage("Clone Repo"){
            steps {
                sh "git clone https://github.com/subbasri17/Jenkins-Practice.git"
            }
        }
        stage("Build"){
            steps {
                dir("Jenkins-Practice") {
                    sh "mvn clean install"
                }
            }
        }
        stage("Test"){
            steps {
                dir("Jenkins-Practice") {
                    sh "mvn test"
                }
            }
        }  
    }
}
